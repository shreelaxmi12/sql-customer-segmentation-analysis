CREATE DATABASE retail_project;
USE retail_project;

CREATE TABLE retail_sales (
    Order_ID VARCHAR(50),
    Order_Date DATE,
    Customer_ID VARCHAR(50),
    Customer_Name VARCHAR(100),
    Region VARCHAR(50),
    State VARCHAR(50),
    Product_Category VARCHAR(50),
    Sub_Category VARCHAR(50),
    Sales FLOAT,
    Profit FLOAT,
    Quantity INT
);

SELECT * FROM retail_sales LIMIT 10;

Total Business Performance
SELECT 
    SUM(Sales) AS Total_Sales,
    SUM(Profit) AS Total_Profit
FROM retail_sales;

Top 10 Customers
SELECT 
    Customer_ID,
    Customer_Name,
    SUM(Sales) AS Total_Sales
FROM retail_sales
GROUP BY Customer_ID, Customer_Name
ORDER BY Total_Sales DESC
LIMIT 10;

Sales by Region
SELECT 
    Region,
    SUM(Sales) AS Total_Sales
FROM retail_sales
GROUP BY Region;

Monthly Sales Trend
SELECT 
    YEAR(Order_Date) AS Year,
    MONTH(Order_Date) AS Month,
    SUM(Sales) AS Monthly_Sales
FROM retail_sales
GROUP BY Year, Month
ORDER BY Year, Month;

Customer Segmentation
SELECT 
    Customer_ID,
    Customer_Name,
    SUM(Sales) AS Total_Sales,
    CASE 
        WHEN SUM(Sales) > 200000 THEN 'High Value'
        WHEN SUM(Sales) BETWEEN 100000 AND 200000 THEN 'Medium Value'
        ELSE 'Low Value'
    END AS Customer_Segment
FROM retail_sales
GROUP BY Customer_ID, Customer_Name;
